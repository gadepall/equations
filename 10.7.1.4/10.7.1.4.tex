This is Problem-4 from Exercise 7.1
\begin{enumerate}
\item Check whether (5, – 2), (6, 4) and (7, – 2) are the vertices of an isosceles triangle. \\
\solution:\\
Given,
\begin{align}
\vec{A}&=\myvec{5\\-2}\\
\vec{B}&=\myvec{6\\4}\\
\vec{C}&=\myvec{7\\-2}\\
\vec{AB}&=\myvec{1\\6}\\
\norm{AB}&=\sqrt{(1)^2+(6)^2}\\
&=\sqrt{37}\\
\norm{BC}&=\sqrt{(1)^2+(-6)^2}\\
&=\sqrt{37}\\
\norm{CA}&=\sqrt{(2)^2+0}\\
&=\sqrt{4}\\
\end{align}
since,$|AB|$=$|BC|$=$\sqrt{37}$,they are isosceles triangle
\end{enumerate}
