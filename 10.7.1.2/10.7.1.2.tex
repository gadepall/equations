\item Find the distance between the following pairs of points.\\
 ii)(0,0),(36, 15).\\
\solution:\\
Given,
\begin{align}
\vec{A}&=\myvec{0\\0}\\
\vec{B}&=\myvec{36\\15}\\
\vec{AB}&=\myvec{36\\15}\\
\norm{AB}&=\sqrt{(36)^2+(15)^2}\\
\norm{AB}&=\sqrt{1296+225}\\
\norm{AB}&=\sqrt{1521}\\
\norm{AB}&=39\\
\end{align}
\end{enumerate}
