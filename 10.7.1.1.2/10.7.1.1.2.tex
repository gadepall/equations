\begin{enumerate}
\item Find the distance between the following pairs of points.\\
 ii)(– 5, 7),(– 1, 3).\\
\solution:\\
Given,
\begin{align}
\vec{A}&=\myvec{-5\\7}\\
\vec{B}&=\myvec{-1\\3}\\
\vec{AB}&=\myvec{4\\-4}\\
\norm{AB}&=\sqrt{(4)^2+(-4)^2}\\
\norm{AB}&=\sqrt{16+16}\\
\norm{AB}&=\sqrt{32}\\
\norm{AB}&=2\sqrt{4}\\
\end{align}
\end{enumerate}
